//NRO-151: Customize the appearance of an NPC sprite
function	script	customizeNPCSprite	{
	setunitdata getnpcid(0), UNPC_CLASS, getarg(0);
	setunitdata getnpcid(0), UNPC_SEX, getarg(1); //if it exists
	setunitdata getnpcid(0), UNPC_HAIRCOLOR, getarg(2);
	setunitdata getnpcid(0), UNPC_HAIRSTYLE, getarg(3);
	setunitdata getnpcid(0), UNPC_HEADTOP, getarg(4);
	setunitdata getnpcid(0), UNPC_HEADMIDDLE, getarg(5); 
	setunitdata getnpcid(0), UNPC_HEADBOTTOM, getarg(6);
	setunitdata getnpcid(0), UNPC_ROBE, getarg(7);
}
